## Overview

The Command component provides a powerful search interface with fuzzy matching and keyboard navigation. It's perfect for command palettes, search interfaces, and quick navigation systems, offering fast filtering and an intuitive user experience.

## Features

- **Fast Search**: Optimized search with fuzzy matching and filtering
- **Keyboard Navigation**: Full keyboard support with arrow keys and shortcuts
- **Composable**: Modular components for flexible layouts
- **Dialog Support**: Built-in command dialog for modal interfaces
- **Accessibility**: ARIA attributes and screen reader support
- **Customizable**: Easy to style and extend with additional functionality
- **Empty States**: Built-in empty state handling
- **Grouping**: Organize commands into logical groups with separators

### Command Palette

```tsx
import { useState } from "react"
import {
  CommandDialog,
  CommandEmpty,
  CommandGroup,
  CommandInput,
  CommandItem,
  CommandList,
  CommandSeparator,
  CommandShortcut,
} from "@/components/ui/command"

function CommandPalette() {
  const [open, setOpen] = useState(false)

  // Toggle with Cmd+K
  useEffect(() => {
    const down = (e: KeyboardEvent) => {
      if (e.key === "k" && (e.metaKey || e.ctrlKey)) {
        e.preventDefault()
        setOpen((open) => !open)
      }
    }
    document.addEventListener("keydown", down)
    return () => document.removeEventListener("keydown", down)
  }, [])

  return (
    <CommandDialog open={open} onOpenChange={setOpen}>
```
