# Chart

A collection of chart components built on top of Recharts for creating beautiful and interactive data visualizations.

## Overview

The Chart components provide a comprehensive set of tools for building charts and data visualizations in React applications. Built on top of Recharts, these components offer a consistent API and styling that integrates seamlessly with the shadcn/ui design system.

## Features

- **Multiple Chart Types**: Line, Bar, Area, Pie, and Radial Bar charts
- **Responsive Design**: Charts automatically adapt to container size
- **Interactive Tooltips**: Rich tooltip content with customizable formatting
- **Legends**: Configurable legends with consistent styling
- **Accessibility**: Built-in accessibility features and keyboard navigation
- **Theming**: Consistent with shadcn/ui color system and CSS variables
- **TypeScript Support**: Full TypeScript support with proper type definitions

## Installation

```bash
npx shadcn@latest add chart
```

## Basic Usage

### Simple Line Chart

```tsx
import {
  ChartConfig,
  ChartContainer,
  ChartTooltip,
  ChartTooltipContent,
} from "@/components/ui/chart"
import { LineChart, Line, XAxis, YAxis, CartesianGrid } from "recharts"

  { month: "Mar", sales: 237 },
  { month: "Apr", sales: 73 },
  { month: "May", sales: 209 },
  { month: "Jun", sales: 214 },
]

const chartConfig = {
  sales: {
    label: "Sales",
    color: "#2563eb",
  },
} satisfies ChartConfig

export function SimpleLineChart() {
  return (
    <ChartContainer config={chartConfig} className="h-[300px]">
      <LineChart data={data}>
        <CartesianGrid strokeDasharray="3 3" />
        <XAxis dataKey="month" />
        <YAxis />
        <ChartTooltip content={<ChartTooltipContent />} />
        <Line 
          type="monotone" 
          dataKey="sales" 
          stroke="var(--color-sales)"
```


## API Reference

### ChartContainer

The main container component that provides configuration and responsive behavior.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `config` | `ChartConfig` | - | Chart configuration object |
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | Chart components |

### ChartConfig

Configuration object that defines chart styling and labels.

```tsx
type ChartConfig = {
  [key: string]: {
    label: string
    color: string
    icon?: React.ComponentType
  }
}
```

### ChartTooltip

Wrapper component for chart tooltips.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `content` | `ReactNode` | - | Tooltip content component |
| `cursor` | `boolean \| object` | `true` | Cursor configuration |
| `active` | `boolean` | - | Whether tooltip is active |

### ChartTooltipContent

Pre-styled tooltip content component.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `active` | `boolean` | - | Whether tooltip is active |
| `payload` | `any[]` | - | Tooltip data payload |
| `label` | `string` | - | Tooltip label |
| `labelKey` | `string` | - | Key for label formatting |
| `nameKey` | `string` | - | Key for name formatting |

### ChartLegend

Wrapper component for chart legends.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `content` | `ReactNode` | - | Legend content component |
| `verticalAlign` | `'top' \| 'middle' \| 'bottom'` | `'bottom'` | Vertical alignment |
| `align` | `'left' \| 'center' \| 'right'` | `'center'` | Horizontal alignment |

### ChartLegendContent

Pre-styled legend content component.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `payload` | `any[]` | - | Legend data payload |
| `nameKey` | `string` | - | Key for name formatting |

### Real-time Chart with Updates

```tsx
import { useState, useEffect } from "react"
import {
  ChartConfig,
  ChartContainer,
  ChartTooltip,
  ChartTooltipContent,
} from "@/components/ui/chart"
import { LineChart, Line, XAxis, YAxis, CartesianGrid } from "recharts"

const chartConfig = {
  value: {
    label: "Value",
    color: "#2563eb",
  },
} satisfies ChartConfig

export function RealTimeChart() {
  const [data, setData] = useState([
    { time: "00:00", value: 100 },
    { time: "00:01", value: 120 },
    { time: "00:02", value: 110 },
  ])

  useEffect(() => {
    const interval = setInterval(() => {
      setData(prevData => {
        const newTime = new Date().toLocaleTimeString('en-US', { 
          hour12: false, 
          hour: '2-digit', 
          minute: '2-digit' 
        })
        const newValue = Math.floor(Math.random() * 100) + 50
        
        const newData = [...prevData, { time: newTime, value: newValue }]
        
        // Keep only last 10 data points
        return newData.slice(-10)
      })
    }, 2000)

    return () => clearInterval(interval)
  }, [])

  return (
    <ChartContainer config={chartConfig} className="h-[300px]">
      <LineChart data={data}>
        <CartesianGrid strokeDasharray="3 3" />
        <XAxis dataKey="time" />
        <YAxis />
        <ChartTooltip content={<ChartTooltipContent />} />
        <Line 
          type="monotone" 
          dataKey="value" 
          stroke="var(--color-value)" 
          strokeWidth={2}
          dot={{ fill: "var(--color-value)" }}
        />
      </LineChart>
    </ChartContainer>
  )
}
```

### Custom Tooltip

```tsx
import {
  ChartConfig,
  ChartContainer,
  ChartTooltip,
} from "@/components/ui/chart"
import { LineChart, Line, XAxis, YAxis, CartesianGrid } from "recharts"

const data = [
  { month: "Jan", sales: 4000, profit: 2400 },
  { month: "Feb", sales: 3000, profit: 1398 },
  { month: "Mar", sales: 2000, profit: 9800 },
]

const chartConfig = {
  sales: {
    label: "Sales",
    color: "#2563eb",
  },
  profit: {
    label: "Profit",
    color: "#60a5fa",
  },
} satisfies ChartConfig

function CustomTooltip({ active, payload, label }: any) {
  if (active && payload && payload.length) {
    return (
      <div className="bg-background border rounded-lg shadow-lg p-3">
        <p className="font-medium">{`Month: ${label}`}</p>
        {payload.map((entry: any, index: number) => (
          <p key={index} style={{ color: entry.color }}>
            {`${entry.name}: $${entry.value.toLocaleString()}`}
          </p>
        ))}
        <p className="text-sm text-muted-foreground mt-1">
          Profit Margin: {((payload[1]?.value / payload[0]?.value) * 100).toFixed(1)}%
        </p>
      </div>
    )
  }
  return null
}

export function ChartWithCustomTooltip() {
  return (
    <ChartContainer config={chartConfig} className="h-[300px]">
      <LineChart data={data}>
        <CartesianGrid strokeDasharray="3 3" />
        <XAxis dataKey="month" />
        <YAxis />
        <ChartTooltip content={<CustomTooltip />} />
        <Line type="monotone" dataKey="sales" stroke="var(--color-sales)" />
        <Line type="monotone" dataKey="profit" stroke="var(--color-profit)" />
      </LineChart>
    </ChartContainer>
  )
}
```

## Best Practices

### Chart Selection

- **Line Charts**: Use for showing trends over time
- **Bar Charts**: Use for comparing categories or discrete values
- **Area Charts**: Use for showing cumulative values or parts of a whole over time
- **Pie Charts**: Use for showing proportions (limit to 5-7 categories)
- **Radial Bar Charts**: Use for showing progress or performance metrics

### Data Formatting

```tsx
// Format large numbers
const formatValue = (value: number) => {
  if (value >= 1000000) {
    return `${(value / 1000000).toFixed(1)}M`
  }
  if (value >= 1000) {
    return `${(value / 1000).toFixed(1)}K`
  }
  return value.toString()
}

// Format percentages
const formatPercentage = (value: number) => `${value.toFixed(1)}%`

// Format currency
const formatCurrency = (value: number) => 
  new Intl.NumberFormat('en-US', {
    style: 'currency',
    currency: 'USD'
  }).format(value)
```

## Integration with Forms

```tsx
import { useForm } from "react-hook-form"
import { Button } from "@/components/ui/button"
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select"

export function InteractiveChart() {
  const [chartType, setChartType] = useState<'line' | 'bar' | 'area'>('line')
  const [timeRange, setTimeRange] = useState<'7d' | '30d' | '90d'>('30d')

  const filteredData = useMemo(() => {
    // Filter data based on time range
    return data.slice(-parseInt(timeRange))
  }, [data, timeRange])

  const renderChart = () => {
    const commonProps = {
      data: filteredData,
      children: [
        <CartesianGrid key="grid" strokeDasharray="3 3" />,
        <XAxis key="x" dataKey="date" />,
        <YAxis key="y" />,
        <ChartTooltip key="tooltip" content={<ChartTooltipContent />} />,
      ]
    }

    switch (chartType) {
      case 'line':
        return (
          <LineChart {...commonProps}>
            {commonProps.children}
            <Line type="monotone" dataKey="value" stroke="var(--color-value)" />
          </LineChart>
        )
      case 'bar':
        return (
          <BarChart {...commonProps}>

            {commonProps.children}
            <Area type="monotone" dataKey="value" stroke="var(--color-value)" fill="var(--color-value)" />
          </AreaChart>
        )
    }
  }

  return (
    <div className="space-y-4">
      <div className="flex gap-4">
        <Select value={chartType} onValueChange={setChartType}>
          <SelectTrigger className="w-[180px]">
            <SelectValue placeholder="Chart type" />

            <SelectItem value="line">Line Chart</SelectItem>
            <SelectItem value="bar">Bar Chart</SelectItem>
```
